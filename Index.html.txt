const docFileCache = {}; // pupilId -> {dataUrl, name}

function toggleDocs(pupilId){
  const panel = document.getElementById(`docPanel_${pupilId}`);
  if(!panel) return;
  panel.style.display = panel.style.display==='none' ? 'block' : 'none';
}

function previewDocUpload(input, pupilId){
  const file = input.files[0]; if(!file) return;
  if(file.size > 1100000){ toast(`File too large — compress below 1MB`); input.value=''; return; }
  const preview = document.getElementById(`docFilePreview_${pupilId}`);
  const reader = new FileReader();
  reader.onload = e=>{
    docFileCache[pupilId] = { dataUrl: e.target.result, name: file.name };
    if(preview) preview.textContent = `✓ Ready: ${file.name} (${(file.size/1024).toFixed(0)}kb)`;
    // Auto-fill name field if empty
    const nameEl = document.getElementById(`docName_${pupilId}`);
    if(nameEl && !nameEl.value) nameEl.value = file.name.replace(/\.[^.]+$/,'');
  };
  reader.readAsDataURL(file);
}

function saveDoc(pupilId){
  const name = document.getElementById(`docName_${pupilId}`)?.value?.trim();
  if(!name){ toast('Add a document name'); return; }
  const category = document.getElementById(`docCat_${pupilId}`)?.value || 'other';
  const date = document.getElementById(`docDate_${pupilId}`)?.value || '';
  const url = document.getElementById(`docUrl_${pupilId}`)?.value?.trim() || '';
  const cached = docFileCache[pupilId];

  if(!cached && !url){ toast('Upload a file or paste a cloud link'); return; }

  if(!db.pupilDocs) db.pupilDocs = [];
  db.pupilDocs.push({
    id: Date.now().toString(),
    pupilId, name, category, date, url,
    dataUrl: cached?.dataUrl || null,
    fileName: cached?.name || null,
    addedAt: new Date().toISOString()
  });
  save();

  // Clear
  delete docFileCache[pupilId];
  const fileInput = document.getElementById(`docFile_${pupilId}`);
  if(fileInput) fileInput.value = '';
  const urlInput = document.getElementById(`docUrl_${pupilId}`);
  if(urlInput) urlInput.value = '';
  const nameInput = document.getElementById(`docName_${pupilId}`);
  if(nameInput) nameInput.value = '';
  const preview = document.getElementById(`docFilePreview_${pupilId}`);
  if(preview) preview.textContent = '';

  toast('Document saved');
  renderPupilList();
  // Re-open the panel after re-render
  setTimeout(()=>{ const p=document.getElementById(`docPanel_${pupilId}`); if(p) p.style.display='block'; }, 50);
}

function deleteDoc(pupilId, docId){
  if(!confirm('Remove this document?')) return;
  db.pupilDocs = (db.pupilDocs||[]).filter(d=>!(d.pupilId==pupilId && d.id==docId));
  save(); renderPupilList();
  setTimeout(()=>{ const p=document.getElementById(`docPanel_${pupilId}`); if(p) p.style.display='block'; }, 50);
}

function buildChips(){
  Object.entries(CHIPS).forEach(([type,items])=>{
    const id={positive:'chipsPos',negative:'chipsNeg',neutral:'chipsNeu'}[type];

    items.forEach(label=>{

        else selectedChips=selectedChips.filter(c=>c.label!==label);
      };
      c.appendChild(ch);
    });
  });
}

  // Score subscales
  const scales={emotional:0,conduct:0,hyper:0,peer:0,prosocial:0};
  SDQ_ITEMS.forEach((item,i)=>{ scales[item.scale]+= item.rev?2-resp[i]:resp[i]; });
  const total=scales.emotional+scales.conduct+scales.hyper+scales.peer;
  db.psychometrics.push({id:Date.now(),measure:'SDQ',pupilId:pid,pupilName:document.getElementById('sdqPupil').options[document.getElementById('sdqPupil').selectedIndex].text,timepoint:v('sdqTP','sel'),informant:v('sdqInformant','sel'),responses:resp,scores:{...scales,totalDifficulties:total},savedAt:new Date().toISOString()});

  document.querySelectorAll('.psych-opt[data-q^="sdq"]').forEach(b=>b.classList.remove('selected'));
}

function calcTier(i){
  const m=BENCH_MOVEMENTS[i];
  const raw=document.getElementById(`bench${i}`).value;
  const cell=document.getElementById(`benchTier${i}`);
  if(!raw){cell.innerHTML='<span class="bench-tier" style="color:var(--grey);border-color:var(--grey)">—</span>';return;}
  const TIER_NAMES=['White','Blue','Purple','Brown','Gold'];
  const TIER_COLORS=['#f8f6f0','#3498db','#9b59b6','#a04000','#C49A2A'];
  let val=parseFloat(raw);
  let tier=0;
  if(!m.lowerIsBetter){ for(let t=0;t<5;t++){if(val>=m.tiers[t])tier=t;} }
  else{ tier=4; for(let t=0;t<5;t++){if(val<=m.tiers[t]){tier=t;break;}} }
  cell.innerHTML=`<span class="bench-tier" style="color:${TIER_COLORS[tier]};border-color:${TIER_COLORS[tier]}">${TIER_NAMES[tier]}</span>`;
}

const pupilPhotos = {}; // pupilId -> [{dataUrl, name}]
const linkCounts = {};  // pupilId -> count

function handlePhotoUpload(input, pupilId){
  const files = [...input.files];
  if(!pupilPhotos[pupilId]) pupilPhotos[pupilId] = [];
  const preview = document.getElementById(`curPhotoPreview_${pupilId}`);
  // Check media consent
  const hasConsent = checkMediaConsent(pupilId);
  if(!hasConsent){
    const pupil = db.pupils.find(p=>p.id==pupilId);
    const warn = document.getElementById(`mediaWarn_${pupilId}`);
    if(warn) warn.style.display='block';
    toast(`⚠ No photo consent (A1) for ${pupil?.first||'this pupil'} — for ASDAN evidence only, not publication`);
  }
  files.forEach(file=>{
    if(file.size > 800000){ toast(`${file.name} too large — compress below 800kb`); return; }
    const reader = new FileReader();
    reader.onload = e=>{
      const dataUrl = e.target.result;
      pupilPhotos[pupilId].push({ dataUrl, name:file.name, noPublishConsent:!hasConsent });
      const thumb = document.createElement('div');
      thumb.className = 'file-thumb';
      thumb.innerHTML = `<img src="${dataUrl}" alt="${file.name}">${!hasConsent?'<span style="color:#eb7e74;font-size:0.7rem">⚠ Evidence only</span>':''}<span>${file.name}</span>`;
      if(preview) preview.appendChild(thumb);
    };
    reader.readAsDataURL(file);
  });
}

function addLinkField(pupilId){
  if(!linkCounts[pupilId]) linkCounts[pupilId] = 1;
  else linkCounts[pupilId]++;
  const n = linkCounts[pupilId];
  const container = document.getElementById(`curLinks_${pupilId}`);
  if(!container) return;
  const row = document.createElement('div');
  row.className = 'drive-link-row';
  row.style.marginTop = '0.4rem';
  row.innerHTML = `<input type="text" id="curLink${n}_${pupilId}" placeholder="Paste Google Drive / OneDrive link">
    <input type="text" id="curLinkLabel${n}_${pupilId}" placeholder="Label" style="max-width:220px">`;
  container.appendChild(row);
}

function collectLinks(pupilId){
  const links = [];
  const n = linkCounts[pupilId]||0;
  for(let i=0;i<=n;i++){
    const url = document.getElementById(`curLink${i}_${pupilId}`)?.value?.trim();
    const label = document.getElementById(`curLinkLabel${i}_${pupilId}`)?.value?.trim();
    if(url) links.push({ url, label: label||url });
  }
  return links;
}

// Get all sessions where a pupil has evidence for a specific criterion
function getCritEvidence(pupilId, criterion){
  const sessions = [];
  db.curriculum.forEach(sess=>{
    const rec = sess.pupilRecords.find(r=>r.pupilId==pupilId);
    if(!rec) return;
    if(rec.criteriaEvidenced.some(e=>e.criterion===criterion)){
      sessions.push({ sessionNum:sess.sessionNum, date:sess.date, observation:rec.observation, goal:rec.goal, reflection:rec.reflection });
    }
  });
  return sessions;
}

// A criterion is "met" when it has at least one specific, traceable evidence record
// with a non-vague observation. We flag quality issues rather than block.
function critIsMet(pupilId, criterion){
  const ev = getCritEvidence(pupilId, criterion);
  if(!ev.length) return { met:false, sessions:[], quality:'none' };
  const hasObs = ev.some(e=>e.observation && e.observation.length > 20);
  const hasRecurrence = ev.length >= 2;
  const quality = hasObs && hasRecurrence ? 'strong' : hasObs ? 'adequate' : 'weak';
  return { met:true, sessions:ev, quality };
}

function getConsentStatus(pupilId){
const c = db.consent.find(x=>x.pupilId==pupilId);
if(!c) return { f1:‘missing’, f2:‘missing’, f3:‘missing’, canAttend:false, medFlags:[], hasMedFlag:false, mediaConsents:[false,false,false,false] };
const f1 = c.f1Returned===‘yes’ && c.f1Consents?.slice(0,2).every(Boolean) ? ‘complete’ : c.f1Returned===‘yes’?‘partial’:‘missing’;
const f2 = !c.f2Status||c.f2Status===‘not_returned’ ? ‘missing’ : c.f2Status===‘returned_none’?‘partial’:‘complete’;
const f3 = c.f3Returned===‘yes’ && c.f3Acks?.slice(0,5).every(Boolean) ? ‘complete’ : c.f3Returned===‘yes’?‘partial’:‘missing’;
const medFlags = c.medFlags||[];
return { f1, f2, f3, canAttend:f1===‘complete’&&f3===‘complete’, medFlags, hasMedFlag:medFlags.some(Boolean), medNotes:c.medNotes, mediaConsents:c.mediaConsents||[false,false,false,false], notes:c.notes };
}

const col=pct===100?’#6fcf97’:pct>=50?‘var(–gold)’:’#eb7e74’;
return `<div class="bar-wrap" style="margin-bottom:0.35rem"><div class="bar-label"><span style="font-size:0.75rem">${crit}</span><span style="color:${col};font-size:0.75rem">${met}/${db.pupils.length}</span></div><div class="bar-track" style="height:6px"><div class="bar-fill" style="width:${pct}%;background:${col}"></div></div></div>`;
}).join(’’);
return `<div style="margin-bottom:0.85rem"><div style="font-size:0.78rem;font-weight:600;color:${unit.color};margin-bottom:0.4rem">${unit.name} <span style="color:var(--grey);font-weight:normal">${allMet}/${db.pupils.length} pupils complete</span></div>${critBars}</div>`;

}

function renderDashCohort(){

const total=db.attendance.length||1;
const totalCrits=Object.values(UNITS).reduce((s,u)=>s+u.criteria.length,0);
el.innerHTML=db.pupils.map(p=>{
const att=db.attendance.filter(a=>a.entries.some(e=>e.pupilId==p.id&&e.status===‘present’)).length;
const attPct=Math.round(att/total*100);
const logs=db.logs.filter(l=>l.pupilId==p.id);
const avgEng=logs.length?(logs.reduce((a,l)=>a+parseInt(l.engagement||0),0)/logs.length).toFixed(1):’—’;
const critsMet=Object.values(UNITS).reduce((s,u)=>s+u.criteria.filter(c=>critIsMet(p.id,c).met).length,0);
