// ============================================================ // APP MODULE (NO GLOBALS) // ============================================================ const App = (() => { // --------------------------- // STATE // --------------------------- const DEFAULT_DB = { pupils: [], logs: [], attendance: [], checkins: [], psychometrics: [], physical: [], school: [], scr: [], targets: [], comms: [], incidents: [], curriculum: [], consent: [], pupilDocs: [] }; let db = loadDB(); const docFileCache = {}; const pupilPhotos = {}; // --------------------------- // UTILITIES // --------------------------- function loadDB() { try { const parsed = JSON.parse(localStorage.getItem("urData") || "{}"); return { ...DEFAULT_DB, ...parsed }; } catch { return { ...DEFAULT_DB }; } } function saveDB() { localStorage.setItem("urData", JSON.stringify(db)); } function getEl(id) { return document.getElementById(id) || null; } function getVal(id) { const el = getEl(id); return el ? el.value.trim() : ""; } function showToast(msg) { if (typeof toast === "function") toast(msg); else console.log(msg); } function validateRequired(fields) { for (const field of fields) { if (!field.value) { showToast(field.message); return false; } } return true; } function createEl(tag, text) { const el = document.createElement(tag); if (text) el.textContent = text; return el; } // --------------------------- // DOCUMENT MANAGEMENT // --------------------------- function previewDocUpload(input, pupilId) { const file = input?.files?.[0]; if (!file) return; const allowed = ["application/pdf", "image/jpeg", "image/png", "application/msword", "application/vnd.openxmlformats-officedocument.wordprocessingml.document"]; if (!allowed.includes(file.type)) { showToast("Invalid file type"); input.value = ""; return; } if (file.size > 1_000_000) { showToast("File must be under 1MB"); input.value = ""; return; } docFileCache[pupilId] = file; const preview = getEl(`docFilePreview_${pupilId}`); if (preview) { preview.textContent = `Ready: ${file.name}`; } const nameEl = getEl(`docName_${pupilId}`); if (nameEl && !nameEl.value) { nameEl.value = file.name.replace(/\.[^.]+$/, ""); } } function saveDoc(pupilId) { const name = getVal(`docName_${pupilId}`); const category = getVal(`docCat_${pupilId}`) || "other"; const date = getVal(`docDate_${pupilId}`); const url = getVal(`docUrl_${pupilId}`); const file = docFileCache[pupilId]; const isValid = validateRequired([ { value: name, message: "Document name required" }, { value: file || url, message: "Upload file or provide link" } ]); if (!isValid) return; const newDoc = { id: Date.now().toString(), pupilId, name, category, date, url: url || null, fileName: file ? file.name : null, addedAt: new Date().toISOString() }; db.pupilDocs = db.pupilDocs || []; db.pupilDocs.push(newDoc); saveDB(); // clear cache delete docFileCache[pupilId]; ["docFile", "docUrl", "docName"].forEach(prefix => { const el = getEl(`${prefix}_${pupilId}`); if (el) el.value = ""; }); const preview = getEl(`docFilePreview_${pupilId}`); if (preview) preview.textContent = ""; showToast("Document saved"); renderPupilList(); } function deleteDoc(pupilId, docId) { if (!confirm("Remove this document?")) return; db.pupilDocs = (db.pupilDocs || []).filter( d => !(d.pupilId === pupilId && d.id === docId) ); saveDB(); renderPupilList(); } // --------------------------- // PHOTO UPLOAD (SAFE) // --------------------------- function handlePhotoUpload(input, pupilId) { const files = Array.from(input?.files || []); if (!files.length) return; pupilPhotos[pupilId] = pupilPhotos[pupilId] || []; const preview = getEl(`curPhotoPreview_${pupilId}`); for (const file of files) { if (file.size > 800000) { showToast(`${file.name} too large`); return; // FAIL FAST } if (!file.type.startsWith("image/")) { showToast("Invalid image file"); return; } } files.forEach(file => { const url = URL.createObjectURL(file); pupilPhotos[pupilId].push({ url, name: file.name }); if (preview) { const wrapper = createEl("div"); wrapper.className = "file-thumb"; const img = document.createElement("img"); img.src = url; img.alt = file.name; const label = createEl("span", file.name); wrapper.appendChild(img); wrapper.appendChild(label); preview.appendChild(wrapper); } }); } // --------------------------- // PUPILS // --------------------------- function addPupil() { const first = getVal("pFirst"); const last = getVal("pLast"); if (!validateRequired([ { value: first, message: "First name required" }, { value: last, message: "Last name required" } ])) return; db.pupils.push({ id: Date.now(), first, last, code: getVal("pCode") || `UR${(db.pupils.length + 1).toString().padStart(3, "0")}`, added: new Date().toISOString() }); saveDB(); ["pFirst", "pLast", "pCode"].forEach(id => { const el = getEl(id); if (el) el.value = ""; }); renderPupilList(); showToast("Pupil added"); } function renderPupilList() { const container = getEl("pupilList"); if (!container) return; container.textContent = ""; if (!db.pupils.length) { container.textContent = "No pupils added"; return; } db.pupils.forEach(p => { const row = createEl("div"); row.textContent = `${p.first} ${p.last} (${p.code})`; container.appendChild(row); }); } // --------------------------- // EVENTS (EXPLICIT) // --------------------------- function bindEvents() { document.querySelectorAll("[data-action]").forEach(el => { el.addEventListener("click", (e) => { const action = el.dataset.action; if (actions[action]) actions[action](e); }); }); } const actions = { addPupil: () => addPupil() }; // --------------------------- // INIT // --------------------------- function init() { renderPupilList(); bindEvents(); } return { init, previewDocUpload, saveDoc, deleteDoc, handlePhotoUpload }; })(); // ============================================================ // INIT APP // ============================================================ document.addEventListener("DOMContentLoaded", () => { App.init(); });
